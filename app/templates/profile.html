Quick API reference

  {{ settings.BASE_URL }}/api/v1/me

  {{ settings.BASE_URL }}/api/v1/gpus

# Book immediately
curl -X POST \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{"gpu_id": 1, "hours": 4, "note": "training run"}' \
  {{ settings.BASE_URL }}/api/v1/reservations

  -H "Authorization: Bearer <token>" \
